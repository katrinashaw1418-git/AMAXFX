```ts
export const withdrawBank = async (req, res) => {
  const { amount, accountName, bsb, accountNumber } = req.body;
  const userId = req.user.id;

  if (!accountName || !bsb || !accountNumber) {
    throw new Error("Missing required fields");
  }

  const tx = await db.transactions.create({
    userId,
    type: "withdrawal",
    method: "bank",
    amount,
    asset: "AUD",

    status: "pending",
    stage: "review",

    description: JSON.stringify({
      accountName,
      bsb,
      accountNumber
    })
  });

  // deduct immediately (reserved)
  await wallet.debit(userId, amount);

  res.json({ tx });
};
```

## ✅ 5.1 Approve Withdrawal

```ts
POST /admin/transactions/:id/approve
```

```ts
export const approveTransaction = async (req, res) => {
  if (!isAdmin(req)) throw new Error("Unauthorized");

  const { id } = req.params;

  const tx = await db.transactions.findById(id);

  if (!tx || tx.type !== "withdrawal") {
    throw new Error("Invalid transaction");
  }

  await db.transactions.update(id, {
    stage: "approved"
  });

  res.json({ success: true });
};
```

## ✅ 5.4 Fail Transaction (VERY IMPORTANT)

```ts
POST /admin/transactions/:id/fail
```

```ts
export const failTransaction = async (req, res) => {
  if (!isAdmin(req)) throw new Error("Unauthorized");

  const { id } = req.params;

  const tx = await db.transactions.findById(id);

  if (!tx) throw new Error("Not found");

  // refund user if withdrawal failed
  if (tx.type === "withdrawal") {
    await wallet.credit(tx.userId, tx.amount);
  }

  await db.transactions.update(id, {
    status: "failed",
    stage: "failed"
  });

  res.json({ success: true });
};
```
