### 2) Add ledger reconciliation

```ts
async function reconcileWalletBalances(userId: number, asOfDate: Date) {
  const reconstructed = await reconstructWalletBalancesAsOf(userId, asOfDate);
  const currentWallets = await storage.getWallets(userId);

  const tolerance = 1e-6;

  const mismatches = currentWallets
    .map((wallet) => {
      const reconstructedBalance = reconstructed[wallet.currency] ?? 0;
      const currentBalance = Number(wallet.balance ?? 0);
      const delta = currentBalance - reconstructedBalance;

      return {
        currency: wallet.currency,
        reconstructedBalance,
        currentBalance,
        delta,
      };
    })
    .filter((x) => Math.abs(x.delta) > tolerance);

  if (mismatches.length > 0) {
    console.warn("ledger_drift_detected", { userId, asOfDate, mismatches });
  }

  return mismatches;
}
```

### 3) Validate transactions before saving

```ts
function validateTransaction(tx: {
  type: string;
  amount: number;
  fromCurrency?: string | null;
  toCurrency?: string | null;
  exchangeRate?: number | null;
}) {
  if (!Number.isFinite(tx.amount) || tx.amount <= 0) {
    throw new Error("Invalid transaction amount");
  }

  if (!tx.type) {
    throw new Error("Transaction type required");
  }

  if (tx.type === "exchange") {
    if (!tx.fromCurrency || !tx.toCurrency) {
      throw new Error("Exchange requires both currencies");
    }

    if (!Number.isFinite(tx.exchangeRate) || tx.exchangeRate! <= 0) {
      throw new Error("Invalid exchange rate");
    }
  }

  if (tx.fromCurrency === tx.toCurrency) {
    throw new Error("fromCurrency and toCurrency cannot be the same");
  }
}
```
