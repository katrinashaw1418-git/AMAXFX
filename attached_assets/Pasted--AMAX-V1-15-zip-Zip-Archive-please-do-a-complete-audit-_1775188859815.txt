import jwt from "jsonwebtoken";

export function requireAuth(req, res, next) {
  const authHeader = req.headers.authorization;

  if (!authHeader) return res.status(401).json({ error: "No token" });

  const token = authHeader.split(" ")[1];

  try {
    const decoded = jwt.verify(token, process.env.JWT_ACCESS_SECRET);
    req.user = decoded;
    next();
  } catch {
    return res.status(401).json({ error: "Invalid token" });
  }
}

export async function processTransaction(userId: number, data: any) {
  const { fromCurrency, toCurrency, amount, fxRate, idempotencyKey } = data;

  // ✅ IDEMPOTENCY
  const existing = await db.query.transactions.findFirst({
    where: (t, { eq }) => eq(t.idempotencyKey, idempotencyKey),
  });

  if (existing) return existing;

  return db.transaction(async (trx) => {
    const tx = await trx.insert("transactions").values({
      userId,
      ...data,
      status: "completed",
    }).returning();

    // ✅ LEDGER ENTRIES
    await trx.insert(ledger).values([
      {
        userId,
        currency: fromCurrency,
        amount: -amount,
        reference: tx[0].id.toString(),
      },
      {
        userId,
        currency: toCurrency,
        amount: amount * fxRate,
        reference: tx[0].id.toString(),
      },
    ]);

    return tx[0];
  });
}

export function calculateSharpe(returns: number[], rf = 0.04) {
  if (!returns.length) return 0;

  const avg = returns.reduce((a, b) => a + b, 0) / returns.length;
  const vol = Math.sqrt(
    returns.reduce((sum, r) => sum + Math.pow(r - avg, 2), 0) / returns.length
  );

  if (vol === 0) return 0;

  return (avg - rf) / vol;
}

    const tx = await trx.insert(transactions).values({
      userId,

      fromCurrency,
      toCurrency,
      fxRate,
      idempotencyKey,
      status: "completed",
    }).returning();

    await trx.insert(ledger).values([

      },

        currency: toCurrency,
        amount: amount * fxRate,
        reference: tx[0].id.toString(),
      },
    ]);

export async function getBalanceForUpdate(trx, userId, currency) {
  const result = await trx.execute(
    `SELECT SUM(amount) as balance
     FROM ledger
     WHERE user_id=$1 AND currency=$2
     FOR UPDATE`,
    [userId, currency]
  );

  return result.rows[0]?.balance || 0;
}

export async function getPortfolio(userId: number) {
  const balances = await getBalances(userId);
  const prices = await getPrices();

  let total = 0;

  const breakdown = balances.map((b: any) => {
    const value = Number(b.balance) * Number(prices[b.currency] || 1);
    total += value;

    return { ...b, value };
  });

  return { total, breakdown };
}

export function antiReplay(req, res, next) {

export async function processTransaction(userId: number, data: any) {
  return db.transaction(async (trx) => {
    const existing = await trx.query.transactions.findFirst({
      where: (t, { eq }) => eq(t.idempotencyKey, data.idempotencyKey),
    });

    if (existing) return existing;

    const balance = await getBalanceForUpdate(trx, userId, data.fromCurrency);

    if (Number(balance) < Number(data.amount)) {
      throw new Error("Insufficient funds");
    }

    const fxRate = await getFxRate(data.fromCurrency, data.toCurrency);

    const tx = await trx.insert(transactions).values({
      userId,
      amount: data.amount,
      fromCurrency: data.fromCurrency,
      toCurrency: data.toCurrency,
      fxRate,
      idempotencyKey: data.idempotencyKey,
      status: "completed",
    }).returning();

    await trx.insert(ledger).values([
      {
        userId,
        currency: data.fromCurrency,
        amount: -data.amount,
        reference: tx[0].id.toString(),
      },
      {
        userId,
        currency: data.toCurrency,
        amount: data.amount * fxRate,
        reference: tx[0].id.toString(),
      },
    ]);

    return tx[0];
  });
}

export async function getBalanceForUpdate(trx, userId, currency) {
  const result = await trx.execute(
    `SELECT SUM(amount) as balance
     FROM ledger
     WHERE user_id=$1 AND currency=$2
     FOR UPDATE`,
    [userId, currency]
  );

  return result.rows[0]?.balance || 0;
}

export async function getPortfolio(userId: number) {
  const balances = await getBalances(userId);
  const prices = await getPrices();

  let total = 0;

  const breakdown = balances.map((b: any) => {
    const value = Number(b.balance) * Number(prices[b.currency] || 1);
    total += value;
    return { ...b, value };
  });

  return { total, breakdown };
}

export function antiReplay(req, res, next) {
  const ts = req.headers["x-timestamp"];
  if (!ts) return res.status(400).send("Missing timestamp");

  if (Date.now() - Number(ts) > 30000) {
    return res.status(403).send("Expired request");
  }

  next();
}
