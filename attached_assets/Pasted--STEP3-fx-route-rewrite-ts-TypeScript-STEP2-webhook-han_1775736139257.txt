import Decimal from "decimal.js";

async function generateFxQuote(user, from, to, amount) {
  const baseRate = new Decimal(await getBaseRate(from, to))

  const baseSpread = getUserSpread(user)
  const volumeAdj = adjustSpreadForVolume(amount)
  const riskAdj = currencyRiskSpread(`${from}/${to}`)

  const finalSpread = baseSpread + volumeAdj + riskAdj

  const clientRate = baseRate.mul(new Decimal(1).minus(finalSpread))

  const buyAmount = new Decimal(amount).mul(clientRate)

  return {
    baseRate: baseRate.toNumber(),
    clientRate: clientRate.toNumber(),
    spread: finalSpread,
    buyAmount: buyAmount.toNumber(),
  }
}

}
