router.post('/', async (req, res) => {
  const { recipientEmail, walletId, amount } = req.body;
  
  const senderWallet = await UserWallet.findByPk(walletId);
  const recipient = await User.findOne({ where: { email: recipientEmail } });
  if(!recipient) return res.status(404).json({ error: 'Recipient not found' });

  // Ensure both are KYC verified
  if(!senderWallet.user.kycVerified || !recipient.kycVerified)
    return res.status(403).json({ error: 'Both parties must be verified' });

  // Internal ledger transfer
  await sequelize.transaction(async (t) => {
    senderWallet.balance -= amount;
    await senderWallet.save({ transaction: t });

    const recipientWallet = await UserWallet.findOne({ where: { userId: recipient.id, currency: 'ETH' } });
    recipientWallet.balance += amount;
    await recipientWallet.save({ transaction: t });

    // Audit log
    await InternalTransferLog.create({
      senderId: senderWallet.userId,
      recipientId: recipient.id,
      amount,
      currency: 'ETH',
    }, { transaction: t });
  });

  res.json({ success: true, message: 'Transfer complete' });
