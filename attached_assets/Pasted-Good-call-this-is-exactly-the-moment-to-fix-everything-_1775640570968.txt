```ts
import { v4 as uuidv4 } from "uuid";

export const createManualDeposit = async (req, res) => {
  const { amount, method } = req.body;
  const userId = req.user.id;

  const reference = `AMAX-${userId}-${Date.now()}`;

  const tx = await db.transactions.create({
    id: uuidv4(),
    userId,
    type: "deposit",
    method,
    amount,
    asset: "AUD",
    status: "pending",
    reference
  });

  res.json({ tx });
};
```

```ts
export const withdrawBank = async (req, res) => {
  const { amount, accountName, bsb, accountNumber } = req.body;

  if (!accountName || !bsb || !accountNumber) {
    throw new Error("Missing required fields");
  }

  const tx = await db.transactions.create({
    userId: req.user.id,
    type: "withdrawal",
    method: "bank",
    amount,
    status: "pending",
    metadata: {
      accountName,
      bsb,
      accountNumber
    }
  });

  res.json({ tx });
};
```
