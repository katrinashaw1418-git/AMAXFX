// Unified annual return assumptions
function getAnnualReturn(category: string, productName?: string): number {
  const rates = {
    real_estate: 0.11,
    corporate_credit: 0.11,
    venture_capital: 0.18,
    digital_assets:
      productName && typeof productName === "string" && productName.includes("Bitcoin")
        ? 0.60
        : 0.0575,
    default: 0.11,
  };
  return rates[category as keyof typeof rates] || rates.default;
}

function calculateInvestmentPerformance(
  product: any,
  investedAmount: number,
  investmentDate: Date,
  currentDate: Date = new Date()
): { currentValue: number; returnAmount: number; returnPercentage: number } {
  if (!product) {
    return { currentValue: investedAmount, returnAmount: 0, returnPercentage: 0 };
  }

  const daysSinceInvestment = Math.floor(
    (currentDate.getTime() - investmentDate.getTime()) / (1000 * 60 * 60 * 24)
  );

  const annualReturn = getAnnualReturn(product.category, product.name);
  let performanceFactor = 1;

  if (daysSinceInvestment > 0) {
    const timeProgress = daysSinceInvestment / 365;
    performanceFactor = Math.pow(1 + annualReturn, timeProgress);
  }

  performanceFactor = Math.max(0.5, performanceFactor);

  const currentValue = investedAmount * performanceFactor;
  const returnAmount = currentValue - investedAmount;
  const returnPercentage = investedAmount > 0 ? (returnAmount / investedAmount) * 100 : 0;

  return {
    currentValue,
    returnAmount,
    returnPercentage,
  };
}

async function convertToUsd(currency: string, amount: number): Promise<number> {
  if (currency === "USD") return amount;

  const directRate = await storage.getFxRate(currency, "USD");
  if (directRate) {
    return amount * parseFloat(directRate.rate);
  }

  const inverseRate = await storage.getFxRate("USD", currency);
  if (inverseRate) {
    return amount / parseFloat(inverseRate.rate);
  }

  return 0;
}

async function calculatePortfolioTotals(userId: number, evaluationDate: Date = new Date()) {
  const wallets = await storage.getWallets(userId);
  const investments = await storage.getUserInvestments(userId);

  let fiatValue = 0;
  let cryptoValue = 0;
  let stablecoinValue = 0;
  let investmentValue = 0;

  for (const wallet of wallets.filter((w) => w.walletType === "fiat")) {
    fiatValue += await convertToUsd(wallet.currency, parseFloat(wallet.balance));
  }

  for (const wallet of wallets.filter((w) => w.walletType === "crypto")) {
    const balance = parseFloat(wallet.balance);

    if (wallet.currency === "USDT" || wallet.currency === "USDC") {
      stablecoinValue += balance;
      continue;
    }

    const rate = await storage.getFxRate(wallet.currency, "USD");
    if (rate) {
      cryptoValue += balance * parseFloat(rate.rate);
    }
  }

  for (const investment of investments) {
    const product = await storage.getInvestmentProduct(investment.productId);
    if (!product) continue;

    const investedAmount = parseFloat(investment.investedAmount);
    const investmentDate = new Date(investment.investmentDate);
    const performance = calculateInvestmentPerformance(
      product,
      investedAmount,
      investmentDate,
      evaluationDate
    );
    investmentValue += performance.currentValue;
  }

  const totalValue = fiatValue + cryptoValue + stablecoinValue + investmentValue;

  return {
    fiatValue,
    cryptoValue,
    stablecoinValue,
    investmentValue,
    totalValue,
  };
}

app.get("/api/portfolio/history", async (req, res) => {
  try {
    const { timeframe = "1M" } = req.query;
    const userId = 1;

    const endDate = new Date();
    const startDate = new Date();

    switch (timeframe) {
      case "1M":
        startDate.setMonth(startDate.getMonth() - 1);
        break;
      case "3M":
        startDate.setMonth(startDate.getMonth() - 3);
        break;
      case "1Y":
        startDate.setFullYear(startDate.getFullYear() - 1);
        break;
      default:
        startDate.setMonth(startDate.getMonth() - 1);
    }

    const snapshots = await storage.getPortfolioSnapshots(userId, startDate, endDate);
    const currentTotals = await calculatePortfolioTotals(userId, endDate);

    const dataPoints = snapshots.map((snapshot) => ({
      date: new Date(snapshot.snapshotDate).toISOString().split("T")[0],
      value: parseFloat(snapshot.totalValue),
      timestamp: new Date(snapshot.snapshotDate).getTime(),
    }));

    if (dataPoints.length === 0) {
      dataPoints.push(
        {
          date: startDate.toISOString().split("T")[0],
          value: currentTotals.totalValue,
          timestamp: startDate.getTime(),
        },
        {
          date: endDate.toISOString().split("T")[0],
          value: currentTotals.totalValue,
          timestamp: endDate.getTime(),
        }
      );
    } else {
      const lastTimestamp = dataPoints[dataPoints.length - 1].timestamp;
      if (lastTimestamp < endDate.getTime()) {
        dataPoints.push({
          date: endDate.toISOString().split("T")[0],
          value: currentTotals.totalValue,
          timestamp: endDate.getTime(),
        });
      }
    }

    const startValue = dataPoints[0]?.value || currentTotals.totalValue;
    const endValue = dataPoints[dataPoints.length - 1]?.value || currentTotals.totalValue;
    const totalReturn = endValue - startValue;
    const totalReturnPercent = startValue > 0 ? (totalReturn / startValue) * 100 : 0;

    res.json({
      timeframe,
      data: dataPoints,
      currentValue: currentTotals.totalValue,
      totalReturn: totalReturn.toFixed(2),
      totalReturnPercent: totalReturnPercent.toFixed(2),
      startValue: startValue.toFixed(2),
      endValue: endValue.toFixed(2),
      historySource: "snapshots",
    });
  } catch (error) {
    console.error("Portfolio history error:", error);
    res.status(500).json({ error: "Failed to get portfolio history" });
  }
});
