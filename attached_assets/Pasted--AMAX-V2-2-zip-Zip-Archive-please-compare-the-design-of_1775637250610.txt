export const ledger = {
  async credit(userId, amount) {
    const balance = await getBalance(userId);

    await db.ledger.create({
      userId,
      type: "credit",
      amount,
      balanceAfter: balance + amount
    });
  },

  async debit(userId, amount) {
    const balance = await getBalance(userId);

    if (balance < amount) throw new Error("Insufficient funds");

    await db.ledger.create({
      userId,
      type: "debit",
      amount,
      balanceAfter: balance - amount
    });
  }
};
