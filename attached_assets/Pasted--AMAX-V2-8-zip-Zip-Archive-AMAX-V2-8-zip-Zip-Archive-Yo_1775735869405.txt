async function runReconciliation() {
  const users = await db("wallets").select("*");

  for (const wallet of users) {
    const externalBalance = await getAirwallexBalance(
      wallet.custodian_wallet_id
    );

    const internalBalance = wallet.balance;

    const diff = internalBalance - externalBalance;

    await db("reconciliation_logs").insert({
      user_id: wallet.user_id,
      currency: wallet.currency,
      internal_balance: internalBalance,
      external_balance: externalBalance,
      difference: diff,
      status: diff === 0 ? "matched" : "mismatch",
    });

    if (diff !== 0) {
      await handleMismatch(wallet, diff);
    }
  }
}
