# 🔴 PART 5 — RETURN SERIES MUST FAIL ON BAD INPUT

## 📁 `financeCore.ts`

---

### PATCH:

```ts
export function calculateReturnSeries(values: number[], cashflows: number[]) {
  if (!values || values.length < 2) {
    throw new Error("Insufficient data for return calculation");
  }

  if (values.includes(null) || values.includes(undefined)) {
    throw new Error("Invalid portfolio values");
  }

  const returns: number[] = [];

  for (let i = 1; i < values.length; i++) {
    const prev = values[i - 1];
    const curr = values[i];
    const flow = cashflows[i] || 0;

    if (prev <= 0) {
      throw new Error("Invalid previous value");
    }

    const r = (curr - prev - flow) / prev;
    returns.push(r);
  }

  return returns;
}
```
