## 2) `server/routes.ts`

Replace your current `backfillPortfolioHistory()` with this **incremental** version. This addresses the remaining issue that backfill can keep trying to fill every missing day across the full range instead of only filling what is newly needed.

```ts
async function backfillPortfolioHistory(
  userId: number,
  startDate: Date,
  endDate: Date
) {
  const normalizedStart = new Date(startDate);
  normalizedStart.setHours(0, 0, 0, 0);

  const normalizedEnd = new Date(endDate);
  normalizedEnd.setHours(0, 0, 0, 0);

  const existing = await storage.getPortfolioSnapshots(userId, normalizedStart, normalizedEnd);

  if (existing.length === 0) {
    const cursor = new Date(normalizedStart);

    while (cursor <= normalizedEnd) {
      await createPortfolioSnapshotForDate(
        userId,
        new Date(cursor),
        cursor.toDateString() === normalizedEnd.toDateString()
          ? "actual"
          : "historical_estimate"
      );

      cursor.setDate(cursor.getDate() + 1);
    }

    return;
  }

  const sortedExisting = [...existing].sort(
    (a: any, b: any) =>
      new Date(a.snapshotDate).getTime() - new Date(b.snapshotDate).getTime()
  );

  const latestExistingDate = new Date(sortedExisting[sortedExisting.length - 1].snapshotDate);
  latestExistingDate.setHours(0, 0, 0, 0);

  const cursor = new Date(latestExistingDate);
  cursor.setDate(cursor.getDate() + 1);

  while (cursor <= normalizedEnd) {
    await createPortfolioSnapshotForDate(
      userId,
      new Date(cursor),
      cursor.toDateString() === normalizedEnd.toDateString()
        ? "actual"
        : "historical_estimate"
    );

    cursor.setDate(cursor.getDate() + 1);
  }
}
```

If you want it to also fill an **earlier gap** when the requested start date is before the earliest stored snapshot, use this slightly stronger version instead:

```ts
async function backfillPortfolioHistory(
  userId: number,
  startDate: Date,
  endDate: Date
) {
  const normalizedStart = new Date(startDate);
  normalizedStart.setHours(0, 0, 0, 0);

  const normalizedEnd = new Date(endDate);
  normalizedEnd.setHours(0, 0, 0, 0);

  const existing = await storage.getPortfolioSnapshots(userId, normalizedStart, normalizedEnd);

  if (existing.length === 0) {
    const cursor = new Date(normalizedStart);

    while (cursor <= normalizedEnd) {
      await createPortfolioSnapshotForDate(
        userId,
        new Date(cursor),
        cursor.toDateString() === normalizedEnd.toDateString()
          ? "actual"
          : "historical_estimate"
      );

      cursor.setDate(cursor.getDate() + 1);
    }

    return;
  }

  const sortedExisting = [...existing].sort(
    (a: any, b: any) =>
      new Date(a.snapshotDate).getTime() - new Date(b.snapshotDate).getTime()
  );

  const earliestExistingDate = new Date(sortedExisting[0].snapshotDate);
  earliestExistingDate.setHours(0, 0, 0, 0);

  const latestExistingDate = new Date(sortedExisting[sortedExisting.length - 1].snapshotDate);
  latestExistingDate.setHours(0, 0, 0, 0);

  if (normalizedStart < earliestExistingDate) {
    const earlyCursor = new Date(normalizedStart);

    while (earlyCursor < earliestExistingDate) {
      await createPortfolioSnapshotForDate(
        userId,
        new Date(earlyCursor),
        "historical_estimate"
      );

      earlyCursor.setDate(earlyCursor.getDate() + 1);
    }
  }

  const lateCursor = new Date(latestExistingDate);
  lateCursor.setDate(lateCursor.getDate() + 1);

  while (lateCursor <= normalizedEnd) {
    await createPortfolioSnapshotForDate(
      userId,
      new Date(lateCursor),
      lateCursor.toDateString() === normalizedEnd.toDateString()
        ? "actual"
        : "historical_estimate"
    );

    lateCursor.setDate(lateCursor.getDate() + 1);
  }
}
```

## Small optional safety guard

```ts
const dayStart = new Date(snapshotDate);
dayStart.setHours(0, 0, 0, 0);

const dayEnd = new Date(snapshotDate);
dayEnd.setHours(23, 59, 59, 999);

const existingForDay = await storage.getPortfolioSnapshots(userId, dayStart, dayEnd);

if (existingForDay.length > 0) {
  return;
}
```
