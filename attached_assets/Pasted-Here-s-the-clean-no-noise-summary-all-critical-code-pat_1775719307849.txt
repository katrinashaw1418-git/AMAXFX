# ✅ PATCH 2 — ENFORCE KYC REFRESH (CRITICAL)

## Problem

`kycRefreshDue` not enforced

## Fix (`auth.ts`)

```ts
export function requireKyc(user: User) {
  if (user.kycStatus !== "verified") {
    throw new Error("KYC verification required");
  }

  // ✅ ADD THIS BLOCK
  if (user.kycRefreshDue && new Date(user.kycRefreshDue) < new Date()) {
    throw new Error(
      "Your KYC review is overdue. Please complete it before continuing."
    );
  }
}
```

## Fix (backend: `routes.ts`)

```ts
app.post("/api/deposit", async (req, res) => {
  const {
    currency,
    amount,
    payerName,
    payerBsb,
    payerAccountNumber,
    payerPayId,
    depositMethod
  } = req.body;

  const tx = await db.transactions.create({
    userId: req.user.id,
    type: "deposit",
    currency,
    amount,
    metadata: {
      payerName,
      payerBsb,
      payerAccountNumber,
      payerPayId,
      depositMethod
    }
  });

  res.json(tx);
});
```
