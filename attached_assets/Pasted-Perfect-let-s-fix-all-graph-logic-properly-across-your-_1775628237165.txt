# 🔧 PATCH 1 — Create a Single Source of Truth

## 📍 New file: `client/src/utils/graph-calculations.ts`

```ts
import Decimal from "decimal.js";

export const toDecimal = (v: any) => new Decimal(v ?? 0);

// Convert any transaction to base currency (AUD)
export const normalizeToAUD = (t: any) => {
  const amount = toDecimal(t.amount);

  // If already AUD
  if (!t.exchangeRate) return amount;

  return amount.mul(toDecimal(t.exchangeRate));
};
```

## 📍 Replace your graph logic with:

```ts
import { normalizeToAUD } from "@/utils/graph-calculations";

export const buildPortfolioSeries = (transactions: any[]) => {
  let running = new Decimal(0);

  return transactions
    .sort((a, b) => new Date(a.createdAt).getTime() - new Date(b.createdAt).getTime())
    .map((t) => {
      const value = normalizeToAUD(t);

      if (t.type === "deposit") running = running.plus(value);
      if (t.type === "withdraw") running = running.minus(value);
      if (t.type === "fee") running = running.minus(value);

      return {
        date: t.createdAt,
        value: running.toNumber()
      };
    });
};
```
