# 🧩 2. TRANSACTION CLASSIFICATION ENGINE (MOST IMPORTANT)

You must explicitly tag every transaction.

## Add this function:

```ts
function classifyTransaction(tx) {
  if (tx.internal) return "INTERNAL_TRANSFER";

  if (tx.currencyFrom !== tx.currencyTo) {
    if (isCrypto(tx.currencyFrom) || isCrypto(tx.currencyTo)) {
      return "CRYPTO_EXCHANGE";
    }
    return "FX_CONVERSION";
  }

  if (tx.crossBorder) return "REMITTANCE";

  if (tx.deposit) return "DEPOSIT";
  if (tx.withdrawal) return "WITHDRAWAL";

  return "UNKNOWN";
}
```

---

## Why this matters

| Type              | Regulatory impact |
| ----------------- | ----------------- |
| INTERNAL_TRANSFER | AML monitoring    |
| FX_CONVERSION     | Remittance rules  |
| REMITTANCE        | IFTI reporting    |
| CRYPTO_EXCHANGE   | DCE registration  |
| WITHDRAWAL        | AML + audit trail |

  if (isHighRiskCountry(tx.destinationCountry)) score += 4;

function complianceCheck(tx, user) {

  const flags = applyAmlRules(tx, user);

  if (risk > 8) {
    return { status: "BLOCK", flags };
  }

  if (flags.length > 0) {
