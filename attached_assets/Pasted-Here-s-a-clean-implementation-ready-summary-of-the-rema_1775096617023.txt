## 🔴 2. FIX: LEDGER RECONCILIATION

```ts
async function reconcileWalletBalances(userId: number, asOfDate: Date) {
  const reconstructed = await reconstructWalletBalancesAsOf(userId, asOfDate);
  const wallets = await storage.getWallets(userId);

  const tolerance = 1e-6;

  const mismatches = wallets
    .map(w => {
      const r = reconstructed[w.currency] ?? 0;
      const c = Number(w.balance ?? 0);
      return { currency: w.currency, delta: c - r };
    })
    .filter(x => Math.abs(x.delta) > tolerance);

  if (mismatches.length) {
    console.warn("ledger_drift_detected", mismatches);
  }

  return mismatches;
}
```

## 🔴 3. FIX: TRANSACTION VALIDATION

```ts
function validateTransaction(tx) {
  if (!Number.isFinite(tx.amount) || tx.amount <= 0) {
    throw new Error("Invalid amount");
  }

  if (!tx.type) {
    throw new Error("Missing transaction type");
  }

  if (tx.type === "exchange") {
    if (!tx.fromCurrency || !tx.toCurrency) {
      throw new Error("Exchange requires currencies");
    }

    if (!Number.isFinite(tx.exchangeRate) || tx.exchangeRate <= 0) {
      throw new Error("Invalid exchange rate");
    }
  }
}
```
