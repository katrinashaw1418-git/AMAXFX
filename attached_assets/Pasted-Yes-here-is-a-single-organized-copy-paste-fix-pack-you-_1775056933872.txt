```ts
async function convertToUsd(currency: string, amount: number): Promise<number> {
  if (currency === "USD") return amount;

  const directRate = await storage.getFxRate(currency, "USD");
  if (directRate) {
    return amount * parseFloat(directRate.rate);
  }

  const inverseRate = await storage.getFxRate("USD", currency);
  if (inverseRate) {
    return amount / parseFloat(inverseRate.rate);
  }

  return 0;
}
```

# 2) `server/routes.ts` — add one valuation function for any date

Paste this near your other portfolio helpers:

```ts
async function calculatePortfolioTotalsAtDate(userId: number, asOfDate: Date) {
  const wallets = await storage.getWallets(userId);
  const investments = await storage.getUserInvestments(userId);

  let fiatValue = 0;
  let cryptoValue = 0;
  let stablecoinValue = 0;
  let investmentValue = 0;

  for (const wallet of wallets) {
    const balance = parseFloat(wallet.balance);

    if (wallet.walletType === "fiat") {
      fiatValue += await convertToUsd(wallet.currency, balance);
      continue;
    }

    if (wallet.currency === "USDT" || wallet.currency === "USDC") {
      stablecoinValue += balance;
      continue;
    }

    const rate = await storage.getFxRate(wallet.currency, "USD");
    if (rate) {
      cryptoValue += balance * parseFloat(rate.rate);
    }
  }

  for (const investment of investments) {
    const product = await storage.getInvestmentProduct(investment.productId);
    if (!product) continue;

    const investedAmount = parseFloat(investment.investedAmount);
    const investmentDate = new Date(investment.investmentDate);

    if (investmentDate <= asOfDate) {
      const performance = calculateInvestmentPerformance(
        product,
        investedAmount,
        investmentDate,
        asOfDate
      );
      investmentValue += performance.currentValue;
    }
  }

  return {
    fiatValue,
    cryptoValue,
    stablecoinValue,
    investmentValue,
    totalValue: fiatValue + cryptoValue + stablecoinValue + investmentValue,
  };
}
```

# 4) `server/routes.ts` — backfill enough history so charts never show null or zero

Paste this helper:

```ts
async function backfillPortfolioHistory(
  userId: number,
  startDate: Date,
  endDate: Date
) {
  const existing = await storage.getPortfolioSnapshots(userId, startDate, endDate);

  const existingDays = new Set(
    existing.map((s: any) => new Date(s.snapshotDate).toISOString().split("T")[0])
  );

  const cursor = new Date(startDate);
  cursor.setHours(0, 0, 0, 0);

  while (cursor <= endDate) {
    const dayKey = cursor.toISOString().split("T")[0];

    if (!existingDays.has(dayKey)) {
      await createPortfolioSnapshotForDate(userId, new Date(cursor), "historical_estimate");
    }

    cursor.setDate(cursor.getDate() + 1);
  }
}
```

app.get("/api/portfolio/history", async (req, res) => {
  try {
    const { timeframe = "1Y" } = req.query;
    const userId = 1;

    const endDate = new Date();
    const startDate = new Date();

    switch (timeframe) {
      case "1W":
        startDate.setDate(startDate.getDate() - 7);
        break;
      case "1M":
        startDate.setMonth(startDate.getMonth() - 1);
        break;
      case "3M":
        startDate.setMonth(startDate.getMonth() - 3);
        break;
      case "6M":
        startDate.setMonth(startDate.getMonth() - 6);
        break;
      case "1Y":
      default:
        startDate.setFullYear(startDate.getFullYear() - 1);
        break;
    }

    await backfillPortfolioHistory(userId, startDate, endDate);

    const snapshots = await storage.getPortfolioSnapshots(userId, startDate, endDate);
    const sortedSnapshots = [...snapshots].sort(
      (a: any, b: any) => new Date(a.snapshotDate).getTime() - new Date(b.snapshotDate).getTime()
    );

    const data = sortedSnapshots.map((snapshot: any) => ({
      date: new Date(snapshot.snapshotDate).toISOString().split("T")[0],
      value: parseFloat(snapshot.totalValue),
      timestamp: new Date(snapshot.snapshotDate).getTime(),
      source: snapshot.source || "historical_estimate",
    }));

    const hasSufficientHistory = data.length >= 2;
    const startValue = hasSufficientHistory ? data[0].value : data.length ? data[0].value : 0;
    const endValue = hasSufficientHistory
      ? data[data.length - 1].value
      : data.length
      ? data[data.length - 1].value
      : 0;

    const totalReturn = hasSufficientHistory ? endValue - startValue : 0;
    const totalReturnPercent =
      hasSufficientHistory && startValue > 0 ? (totalReturn / startValue) * 100 : 0;

    const historySource =
      data.some((d: any) => d.source === "historical_estimate")
        ? "historical_estimate"
        : "historical";

    res.json({
      timeframe,
      data,
      currentValue: endValue,
      totalReturn: totalReturn.toFixed(2),
      totalReturnPercent: totalReturnPercent.toFixed(2),
      hasSufficientHistory,
      historySource,
    });
  } catch (error) {
    console.error("Portfolio history error:", error);
    res.status(500).json({ error: "Failed to get portfolio history" });
  }
});

# 8) `server/routes.ts` — add a historical + forecast chart route

If you already have `/api/portfolio/performance-chart`, replace it with this clean version:

```ts
app.get("/api/portfolio/performance-chart", async (req, res) => {
  try {
    const userId = 1;
    const today = new Date();

    const historyStart = new Date(today);
    historyStart.setFullYear(historyStart.getFullYear() - 1);

    await backfillPortfolioHistory(userId, historyStart, today);

    const snapshots = await storage.getPortfolioSnapshots(userId, historyStart, today);
    const sortedSnapshots = [...snapshots].sort(
      (a: any, b: any) => new Date(a.snapshotDate).getTime() - new Date(b.snapshotDate).getTime()
    );

    const historical = sortedSnapshots.map((s: any) => ({
      date: new Date(s.snapshotDate).toISOString().split("T")[0],
      historical: parseFloat(s.totalValue),
      projected: null,
      source: s.source || "historical_estimate",
    }));

    const lastHistoricalValue =
      historical.length > 0 ? historical[historical.length - 1].historical : 0;

    const annualForecastRate = 0.10;
    const forecastMonths = 12;
    const forecast: any[] = [];

    for (let i = 1; i <= forecastMonths; i++) {
      const date = new Date(today);
      date.setMonth(date.getMonth() + i);

      const years = i / 12;
      const projectedValue = lastHistoricalValue * Math.pow(1 + annualForecastRate, years);

      forecast.push({
        date: date.toISOString().split("T")[0],
        historical: null,
        projected: projectedValue,
        source: "forecast",
      });
    }

    res.json({
      chartSource:
        historical.some((p: any) => p.source === "historical_estimate")
          ? "historical_estimate_plus_forecast"
          : "historical_plus_forecast",
      data: [...historical, ...forecast],
    });
  } catch (error) {
    console.error("Performance chart error:", error);
    res.status(500).json({ error: "Failed to get performance chart" });
  }
});
```

const getReturnFromHistory = (days: number) => {

  const cutoff = Date.now() - days * 24 * 60 * 60 * 1000;

  const endPoint = historyPoints[historyPoints.length - 1];

  if (!startPoint || !endPoint || startPoint.value <= 0) return null;
  return ((endPoint.value - startPoint.value) / startPoint.value) * 100;
};

const periodPerformanceData = periodConfig.map(({ period, days }) => ({
  period,
  portfolioReturn: getReturnFromHistory(days),
