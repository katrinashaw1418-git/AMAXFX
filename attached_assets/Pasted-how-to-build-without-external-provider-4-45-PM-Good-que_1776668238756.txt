app.post("/api/auth/phone/request", async (req, res) => {
  try {
    const { phone, email, firstName, lastName } = req.body;

    if (!phone || !/^\+?[1-9]\d{7,14}$/.test(phone.replace(/[\s\-()]/g, ""))) {
      return res.status(400).json({ error: "Please enter a valid phone number." });
    }
    if (!email || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
      return res.status(400).json({ error: "Please enter a valid email address." });
    }
    if (!firstName?.trim() || !lastName?.trim()) {
      return res.status(400).json({ error: "Please enter your full name." });
    }

    // Normalise phone — strip spaces/dashes, ensure + prefix
    const normalised = "+" + phone.replace(/[\s\-()]/g, "").replace(/^\+/, "");

    // Check if phone already registered
    const [byPhone] = await db.select().from(users)
      .where(eq(users.phoneNumber, normalised));
    if (byPhone) {
      return res.status(409).json({
        error: "A account with this phone number already exists. Please sign in.",
        existingAccount: true,
      });
    }

    // Check if email already registered
    const existing = await storage.getUserByEmail(email.toLowerCase().trim());
    if (existing) {
      return res.status(409).json({
        error: "An account with this email already exists. Please sign in.",
      });
    }

    // Derive unique username
    const base = email.split("@")[0].toLowerCase().replace(/[^a-z0-9_]/g, "_");
    let username = base;
    let attempt = 0;
    while (await storage.getUserByUsername(username)) {
      attempt++;
      username = `${base}${attempt}`;
    }

    // Create account — no password (phone/email auth)
    const newUser = await storage.createUser({
      username,
      email: email.toLowerCase().trim(),
      password: null as any,
      firstName: firstName.trim(),
      lastName: lastName.trim(),
      phoneNumber: normalised,
      kycStatus: "pending",
      userTier: "standard",
    });

    // Create default AUD wallet
    await storage.createWallet({
      userId: newUser.id,
      currency: "AUD",
      balance: "0.00",
      availableBalance: "0.00",
      walletType: "fiat",
    });

    // Generate OTP — reuse exact same mechanism as email registration
    const verifyToken = randomBytes(32).toString("hex");
    const otp = Math.floor(100000 + Math.random() * 900000).toString();
    const expiry = new Date(Date.now() + 10 * 60 * 1000); // 10 minutes

    await db.update(users).set({
      emailVerificationToken: verifyToken,
      emailVerificationTokenExpiry: expiry,
      emailOtp: otp,
      emailVerified: false,
    }).where(eq(users.id, newUser.id));

    // Send OTP via email (same sendVerificationEmail already used elsewhere)
    const baseUrl = `${req.protocol}://${req.get("host")}`;
    await sendVerificationEmail(
      newUser.email,
      newUser.firstName,
      verifyToken,
      otp,
      baseUrl
    ).catch(() => {});

    await writeAuditLog(
      newUser.id, "phone_register_started", "user",
      String(newUser.id), { phone: normalised, email }, req.ip || null
    );

    const token = signToken({
      userId: newUser.id,
      username: newUser.username,
      email: newUser.email,
    });

    res.status(201).json({
      token,
      userId: newUser.id,
      email: newUser.email,
      emailSent: emailConfigured,
      ...(!emailConfigured ? { devOtp: otp } : {}),
    });

    console.error("[phone-request]", err.message);
    if (err.status) return res.status(err.status).json({ error: err.message });
    res.status(500).json({ error: "Registration failed. Please try again." });
  }

export async function sendSmsOtp(
  phone: string,
  otp: string
): Promise<{ sent: boolean }> {
  const apiKey    = process.env.VONAGE_API_KEY;
  const apiSecret = process.env.VONAGE_API_SECRET;

  if (!apiKey || !apiSecret) {
    // Dev fallback — log to console exactly like email OTP does
    console.log(`[sms] OTP for ${phone}: ${otp}`);
    return { sent: false };
  }

  const res = await fetch("https://rest.nexmo.com/sms/json", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      api_key:    apiKey,
      api_secret: apiSecret,
      from:       "AMAX GLOBAL",
      to:         phone.replace(/^\+/, ""),  // Vonage wants no + prefix
      text:       `Your AMAX GLOBAL verification code is: ${otp}. Valid for 10 minutes.`,
    }),
  });

  const data = await res.json() as any;
  const status = data?.messages?.[0]?.status;

  if (status !== "0") {
    console.error("[sms] Vonage error:", data?.messages?.[0]);
    throw new Error("Failed to send SMS. Please check your number and try again.");
  }

  return { sent: true };
}

      return res.status(400).json({ error: "No verification code found. Please request a new one." });
    }
    if (new Date() > new Date(row.expires_at)) {
      return res.status(400).json({ error: "Code expired. Please request a new one." });
    }
    if (code.trim() !== row.otp) {
      return res.status(400).json({ error: "Invalid code. Please try again." });
    }

    // Delete used OTP immediately
    await db.execute(sql`DELETE FROM phone_otps WHERE phone = ${normalised}`);

    // Find or create user by phone number
    const [existing] = await db.select().from(users)
      .where(eq(users.phoneNumber, normalised));

    let user = existing;
    let isNewUser = false;

    if (!user) {
      // New user — derive username from phone
      const suffix = normalised.replace(/\D/g, "").slice(-8);
      let username = `ph${suffix}`;
      let attempt  = 0;
      while (await storage.getUserByUsername(username)) {
        username = `ph${suffix}_${++attempt}`;
      }

      user = await storage.createUser({
        username,
        email: `${normalised.replace(/\D/g, "")}@phone.amax`,
        password: null as any,
        firstName: firstName?.trim() || "AMAX",
        lastName:  lastName?.trim()  || "User",
        phoneNumber: normalised,
        emailVerified: true,
        kycStatus: "pending",
        userTier: "standard",
      });

      await storage.createWallet({
        userId: user.id, currency: "AUD",
        balance: "0.00", availableBalance: "0.00", walletType: "fiat",
      });

      await writeAuditLog(
        user.id, "phone_register", "user", String(user.id),
        { phone: normalised }, req.ip || null
      );
      isNewUser = true;
    } else {
      await writeAuditLog(
        user.id, "phone_login", "user", String(user.id),
        { phone: normalised }, req.ip || null
      );
    }

    if (user.accountFrozen) {
      return res.status(403).json({ error: "Account suspended. Contact support at info@amaxglobal.com.au" });
    }

    const token = signToken({ userId: user.id, username: user.email, email: user.email });
    res.json({ token, isNewUser, user: { id: user.id, email: user.email, firstName: user.firstName } });

  } catch (err: any) {
    console.error("[phone-verify-otp]", err.message);
    res.status(500).json({ error: "Verification failed. Please try again." });
  }
});
